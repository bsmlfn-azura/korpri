⚠️ Konfigurasi Diperlukan

Aplikasi ini membutuhkan Google Apps Script URL untuk berfungsi.

1. Copy kode di bawah ini ke Code.gs di Google Apps Script.
2. Deploy sebagai Web App (Execute as: Me, Who has access: Anyone).
3. Copy Web App URL yang muncul.
4. Buka file index.html ini di text editor, cari variabel GOOGLE_SCRIPT_URL di bagian bawah, dan tempel URL tersebut.

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Data');
  const data = sheet.getDataRange().getValues();
  const headers = data.shift();
  // Mapping headers to keys
  const result = data.map(row => {
    let obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    return obj;
  });
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Data');
  const params = JSON.parse(e.postData.contents);
  sheet.appendRow([
    params.receiptNo, params.printDate, params.depositor, params.opd,
    params.depositDate, params.receiverName, params.periodeBulan, params.periodeTahun,
    params.totalBayar, params.bayar50, JSON.stringify(params.details)
  ]);
  return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
}
